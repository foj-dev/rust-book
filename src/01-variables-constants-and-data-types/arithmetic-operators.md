# Arithmetic Operators

Used for basic mathematical calculations.

`+`

```rust
//
let a = 1;
let b = 1;
println!("a + b = {}", a + b);
```

```rust
//
let a = 2;
let b = 3;
println!("a + b = {}", a + b);
```

`-`

```rust
//
let a = 3;
let b = 3;
println!("a - b = {}", a - b);
```

```rust
//
let a = 4;
let b = 5;
println!("a - b = {}", a - b);
```

`*`

```rust
//
let a = 5;
let b = 6;
println!("a * b = {}", a * b);
```

```rust
//
let a = 6;
let b = 7;
println!("a * b = {}", a * b);
```
    
`/`

```rust
//
let a = 7;
let b = 8;
println!("a / b = {}", a / b);
```

`%`

```rust
//
let a = 4;
let b = 2;
println!("a % b = {}", a % b);
```

```rust
//
let a = 5;
let b = 3;
println!("a % b = {}", a % b);
```
